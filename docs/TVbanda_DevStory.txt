```typescript
// hooks/useHlsPlayer.ts
import Hls from 'hls.js';

export const useHlsPlayer = (videoRef: RefObject<HTMLVideoElement>, src: string) => {
  useEffect(() => {
    if (!videoRef.current) return;

    const video = videoRef.current;
    let hls: Hls | null = null;

    if (Hls.isSupported()) {
      hls = new Hls({
        enableWorker: true,
        lowLatencyMode: true,
        backBufferLength: 90
      });
      hls.loadSource(src);
      hls.attachMedia(video);
    } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
      video.src = src;
    }

    return () => hls?.destroy();
  }, [src]);
};
```
