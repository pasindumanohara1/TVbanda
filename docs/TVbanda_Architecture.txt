## Technical Architecture & System Design

## рҹ“җ System Overview

### Architecture Pattern
**Frontend-Only SPA (Single Page Application)**
- No backend server required
- Client-side data aggregation from iptv-org APIs
- Local storage for user state persistence
- Service Worker for offline capabilities

## рҹ”Ң API Integration

### iptv-org API Endpoints
```typescript
// services/iptvService.ts
const API_BASE = 'https://iptv-org.github.io/api';

class IPTVService {
  private baseUrl = API_BASE;
  private cacheDuration = 24 * 60 * 60 * 1000; // 24 hours

  // Fetch all channels
  async getChannels(): Promise<Channel[]> {
    const cacheKey = 'channels';
    const cached = cacheService.get<Channel[]>(cacheKey);
    if (cached) return cached;

    const response = await fetch(`${this.baseUrl}/channels.json`);
    const data = await response.json();
    cacheService.set(cacheKey, data.channels, this.cacheDuration);
    return data.channels;
  }

  // Fetch all streams
  async getStreams(): Promise<Stream[]> {
    const cacheKey = 'streams';
    const cached = cacheService.get<Stream[]>(cacheKey);
    if (cached) return cached;

    const response = await fetch(`${this.baseUrl}/streams.json`);
    const data = await response.json();
    cacheService.set(cacheKey, data.streams, this.cacheDuration);
    return data.streams;
  }

  // Fetch categories
  async getCategories(): Promise<Category[]> {
    const response = await fetch(`${this.baseUrl}/categories.json`);
    const data = await response.json();
    return data.categories;
  }

  // Fetch countries
  async getCountries(): Promise<Country[]> {
    const response = await fetch(`${this.baseUrl}/countries.json`);
    const data = await response.json();
    return data.countries.map((c: any) => ({
      code: c.code,
      name: c.name,
      flag: getFlagEmoji(c.code)
    }));
  }

  // Merge channels with their streams
  async getChannelsWithStreams(): Promise<ChannelWithStreams[]> {
    const [channels, streams] = await Promise.all([
      this.getChannels(),
      this.getStreams()
    ]);

    const streamsByChannel = streams.reduce((acc, stream) => {
      if (stream.channel) {
        if (!acc[stream.channel]) acc[stream.channel] = [];
        acc[stream.channel].push(stream);
      }
      return acc;
    }, {} as Record<string, Stream[]>);

    return channels.map(channel => ({
      ...channel,
      streams: streamsByChannel[channel.id] || []
    })).filter(c => c.streams.length > 0); // Only channels with streams
  }
}

export const iptvService = new IPTVService();
```

### Caching Strategy
```typescript
// services/cacheService.ts
class CacheService {
  private prefix = 'tvbanda_';

  get<T>(key: string): T | null {
    const item = localStorage.getItem(this.prefix + key);
    if (!item) return null;

    const { value, expiry } = JSON.parse(item);
    if (expiry && Date.now() > expiry) {
      localStorage.removeItem(this.prefix + key);
      return null;
    }
    return value as T;
  }

  set<T>(key: string, value: T, ttl?: number): void {
    const item = {
      value,
      expiry: ttl ? Date.now() + ttl : null
    };
    localStorage.setItem(this.prefix + key, JSON.stringify(item));
  }

  remove(key: string): void {
    localStorage.removeItem(this.prefix + key);
  }

  clear(): void {
    Object.keys(localStorage)
      .filter(key => key.startsWith(this.prefix))
      .forEach(key => localStorage.removeItem(key));
  }
}

export const cacheService = new CacheService();
```

### HLS Player Hook
```typescript
// hooks/useHlsPlayer.ts
import { useEffect, useRef, useCallback, useState } from 'react';
import Hls from 'hls.js';

interface UseHlsPlayerOptions {
  onError?: (error: Error) => void;
  onQualityChange?: (levels: Hls.Level[]) => void;
  autoPlay?: boolean;
}

interface UseHlsPlayerReturn {
  videoRef: React.RefObject<HTMLVideoElement>;
  hls: Hls | null;
  isLoading: boolean;
  isPlaying: boolean;
  currentQuality: number;
  availableQualities: Hls.Level[];
  play: () => void;
  pause: () => void;
  setQuality: (level: number) => void;
  volume: number;
  setVolume: (vol: number) => void;
  toggleMute: () => void;
  isMuted: boolean;
}

export const useHlsPlayer = (
  src: string,
  options: UseHlsPlayerOptions = {}
): UseHlsPlayerReturn => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const hlsRef = useRef<Hls | null>(null);

  const [isLoading, setIsLoading] = useState(true);
  const [isPlaying, setIsPlaying] = useState(false);
  const [currentQuality, setCurrentQuality] = useState(-1); // -1 = auto
  const [availableQualities, setAvailableQualities] = useState<Hls.Level[]>([]);
  const [volume, setVolumeState] = useState(1);
  const [isMuted, setIsMuted] = useState(false);

  useEffect(() => {
    const video = videoRef.current;
    if (!video || !src) return;

    setIsLoading(true);

    if (Hls.isSupported()) {
      const hls = new Hls({
        enableWorker: true,
        lowLatencyMode: true,
        backBufferLength: 90,
        maxBufferLength: 30,
        maxMaxBufferLength: 60,
        manifestLoadingTimeOut: 10000,
        manifestLoadingMaxRetry: 3
      });

      hlsRef.current = hls;

      hls.on(Hls.Events.MANIFEST_PARSED, (_, data) => {
        setAvailableQualities(data.levels);
        setIsLoading(false);
        if (options.autoPlay) {
          video.play().catch(() => setIsPlaying(false));
        }
      });

      hls.on(Hls.Events.LEVEL_SWITCHED, (_, data) => {
        setCurrentQuality(data.level);
      });

      hls.on(Hls.Events.ERROR, (_, data) => {
        if (data.fatal) {
          options.onError?.(new Error(`HLS Error: ${data.type}`));
        }
      });

      hls.loadSource(src);
      hls.attachMedia(video);
    } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
      // Native HLS support (Safari)
      video.src = src;
      video.addEventListener('loadedmetadata', () => {
        setIsLoading(false);
        if (options.autoPlay) video.play();
      });
    }

    return () => {
      hlsRef.current?.destroy();
      hlsRef.current = null;
    };
  }, [src]);

  const play = useCallback(() => {
    videoRef.current?.play();
    setIsPlaying(true);
  }, []);

  const pause = useCallback(() => {
    videoRef.current?.pause();
    setIsPlaying(false);
  }, []);

  const setQuality = useCallback((level: number) => {
    if (hlsRef.current) {
      hlsRef.current.currentLevel = level;
      setCurrentQuality(level);
    }
  }, []);

  const setVolume = useCallback((vol: number) => {
    if (videoRef.current) {
      videoRef.current.volume = vol;
      setVolumeState(vol);
      setIsMuted(vol === 0);
    }
  }, []);

  const toggleMute = useCallback(() => {
    if (videoRef.current) {
      videoRef.current.muted = !isMuted;
      setIsMuted(!isMuted);
    }
  }, [isMuted]);

  return {
    videoRef,
    hls: hlsRef.current,
    isLoading,
    isPlaying,
    currentQuality,
    availableQualities,
    play,
    pause,
    setQuality,
    volume,
    setVolume,
    toggleMute,
    isMuted
  };
};
```

      toggleFavoritesOnly: () => set(state => ({ 
        showFavoritesOnly: !state.showFavoritesOnly 
      })),
      toggleNSFW: () => set(state => ({ hideNSFW: !state.hideNSFW })),

      filteredChannels: () => {
        const state = get();
        return state.channels.filter(channel => {
          // Category filter
          if (state.selectedCategory && 
              !channel.categories.includes(state.selectedCategory)) {
            return false;
          }

          // Country filter
          if (state.selectedCountry && 
              channel.country !== state.selectedCountry) {
            return false;
          }

          // Search filter
          if (state.searchQuery) {
            const query = state.searchQuery.toLowerCase();
            const nameMatch = channel.name.toLowerCase().includes(query);
            const altMatch = channel.alt_names?.some(n => 
              n.toLowerCase().includes(query)
            );
            if (!nameMatch && !altMatch) return false;
          }

          // NSFW filter
          if (state.hideNSFW && channel.is_nsfw) return false;

          return true;
        });
      }
    }),
    {
      name: 'tvbanda-channels',
      partialize: (state) => ({ 
        selectedCategory: state.selectedCategory,
        selectedCountry: state.selectedCountry,
        hideNSFW: state.hideNSFW
      })
    }
  )
);

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface PlayerState {
  currentChannel: ChannelWithStreams | null;
  currentStreamIndex: number;
  isFullscreen: boolean;
  isTheaterMode: boolean;
  volume: number;
  isMuted: boolean;

  setCurrentChannel: (channel: ChannelWithStreams | null) => void;
  setCurrentStreamIndex: (index: number) => void;
  toggleFullscreen: () => void;
  toggleTheaterMode: () => void;
  setVolume: (volume: number) => void;
  toggleMute: () => void;
  nextStream: () => void;
  previousStream: () => void;
}

export const usePlayerStore = create<PlayerState>()(
  persist(
    (set, get) => ({
      currentChannel: null,
      currentStreamIndex: 0,
      isFullscreen: false,
      isTheaterMode: false,
      volume: 1,
      isMuted: false,

      setCurrentChannel: (channel) => set({ 
        currentChannel: channel,
        currentStreamIndex: 0 
      }),

      setCurrentStreamIndex: (index) => set({ currentStreamIndex: index }),

      toggleFullscreen: () => set(state => ({ 
        isFullscreen: !state.isFullscreen 
      })),

      toggleTheaterMode: () => set(state => ({ 
        isTheaterMode: !state.isTheaterMode 
      })),

      setVolume: (volume) => set({ volume: Math.max(0, Math.min(1, volume)) }),

      toggleMute: () => set(state => ({ isMuted: !state.isMuted })),

      nextStream: () => {
        const { currentChannel, currentStreamIndex } = get();
        if (!currentChannel) return;
        const nextIndex = (currentStreamIndex + 1) % currentChannel.streams.length;
        set({ currentStreamIndex: nextIndex });
      },

      previousStream: () => {
        const { currentChannel, currentStreamIndex } = get();
        if (!currentChannel) return;
        const prevIndex = currentStreamIndex === 0 
          ? currentChannel.streams.length - 1 
          : currentStreamIndex - 1;
        set({ currentStreamIndex: prevIndex });
      }
    }),
    {
      name: 'tvbanda-player',
      partialize: (state) => ({ volume: state.volume, isMuted: state.isMuted })
    }
  )
);

## рҹ”§ Service Worker (PWA)

```typescript
// public/sw.js (Vite PWA will generate this, but here's the logic)
const CACHE_NAME = 'tvbanda-v1';
const STATIC_ASSETS = [
  '/',
  '/index.html',
  '/manifest.json',
  '/icons/icon-192x192.png',
  '/icons/icon-512x512.png'
];

// Install event - cache static assets
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(STATIC_ASSETS);
    })
  );
  self.skipWaiting();
});

// Activate event - clean old caches
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames
          .filter((name) => name !== CACHE_NAME)
          .map((name) => caches.delete(name))
      );
    })
  );
  self.clients.claim();
});

// Fetch event - network first, cache fallback for API
self.addEventListener('fetch', (event) => {
  const { request } = event;
  const url = new URL(request.url);

  // API calls - network first, cache fallback
  if (url.hostname === 'iptv-org.github.io') {
    event.respondWith(
      fetch(request)
        .then((response) => {
          const clone = response.clone();
          caches.open(CACHE_NAME).then((cache) => {
            cache.put(request, clone);
          });
          return response;
        })
        .catch(() => caches.match(request))
    );
    return;
  }

  // Static assets - cache first
  event.respondWith(
    caches.match(request).then((response) => {
      return response || fetch(request);
    })
  );
});
```

// Lazy load images with intersection observer
const useLazyImage = (src: string) => {
  const [loaded, setLoaded] = useState(false);
  const imgRef = useRef<HTMLImageElement>(null);

  useEffect(() => {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = new Image();
          img.src = src;
          img.onload = () => setLoaded(true);
          observer.disconnect();
        }
      });
    });

    if (imgRef.current) observer.observe(imgRef.current);
    return () => observer.disconnect();
  }, [src]);

  return { ref: imgRef, loaded };
};
